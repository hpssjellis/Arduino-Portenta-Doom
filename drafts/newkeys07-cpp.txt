static unsigned char convertToDoomKey(unsigned char scancode)
{
    unsigned char key = 0;

    switch (scancode)
    {
    case 0x9C:
    case 0x1C:
        key = KEY_ENTER;
        break;
    case 0x01:
        key = KEY_ESCAPE;
        break;
    case 0xCB:
    case 0x4B:
        key = KEY_LEFTARROW;
        break;
    case 0xCD:
    case 0x4D:
        key = KEY_RIGHTARROW;
        break;
    case 0xC8:
    case 0x48:
        key = KEY_UPARROW;
        break;
    case 0xD0:
    case 0x50:
        key = KEY_DOWNARROW;
        break;
    case 0x1D:
        key = KEY_FIRE;
        break;
    case 0x39:
        key = KEY_USE;
        break;
    case 0x2A:
    case 0x36:
        key = KEY_RSHIFT;
        break;
    case 0x15:
        key = 'y';
        break;
    default:
        break;
    }

    return key;
}

static void addKeyToQueue(int pressed, unsigned char keyCode)
{
	//printf("key hex %x decimal %d\n", keyCode, keyCode);

  unsigned char key = convertToDoomKey(keyCode);

  unsigned short keyData = (pressed << 8) | key;

  s_KeyQueue[s_KeyQueueWriteIndex] = keyData;
  s_KeyQueueWriteIndex++;
  s_KeyQueueWriteIndex %= KEYQUEUE_SIZE;
}

static void InitCLUT(uint32_t * clut)
{
  uint32_t  red = 0, green = 0, blue = 0;
  uint32_t  i = 0x00;

  /* Color map generation */
  for (i = 0; i < 256; i++)
  {
    /* Generate red, green and blue values */
    red = (i * 8) % 256;
    green = (i * 6) % 256;
    blue = (i * 4) % 256;

    red = red << 16;
    green = green << 8;

    /* Store the 32-bit value */
    clut[i] = 0xFF000000 | (red + green + blue);
  }
}

uint32_t __ALIGNED(32) L8_CLUT[256];
static DMA2D_CLUTCfgTypeDef clut;

static void DMA2D_Init(uint16_t xsize, uint16_t ysize)
{
  /*##-1- Configure the DMA2D Mode, Color Mode and output offset #############*/
  DMA2D_Handle.Init.Mode         = DMA2D_M2M_PFC;
  DMA2D_Handle.Init.ColorMode    = DMA2D_OUTPUT_RGB565;
  DMA2D_Handle.Init.OutputOffset = 0; //LCD_X_Size - xsize;
  DMA2D_Handle.Init.AlphaInverted = DMA2D_REGULAR_ALPHA;  /* No Output Alpha Inversion*/
  DMA2D_Handle.Init.RedBlueSwap   = DMA2D_RB_REGULAR;     /* No Output Red & Blue swap */

  /*##-2- DMA2D Callbacks Configuration ######################################*/
  DMA2D_Handle.XferCpltCallback  = NULL;

  /*##-3- Foreground Configuration ###########################################*/
  DMA2D_Handle.LayerCfg[1].AlphaMode = DMA2D_REPLACE_ALPHA; //DMA2D_NO_MODIF_ALPHA;
  DMA2D_Handle.LayerCfg[1].InputAlpha = 0x00;
  DMA2D_Handle.LayerCfg[1].InputColorMode = DMA2D_INPUT_L8; //DMA2D_OUTPUT_RGB565;
  //DMA2D_Handle.LayerCfg[1].ChromaSubSampling = cssMode;
  DMA2D_Handle.LayerCfg[1].InputOffset = 0; //LCD_Y_Size - ysize;
  DMA2D_Handle.LayerCfg[1].RedBlueSwap = DMA2D_RB_REGULAR; /* No ForeGround Red/Blue swap */
  DMA2D_Handle.LayerCfg[1].AlphaInverted = DMA2D_REGULAR_ALPHA; /* No ForeGround Alpha inversion */

  DMA2D_Handle.Instance          = DMA2D;

  /*##-4- DMA2D Initialization     ###########################################*/
  HAL_DMA2D_Init(&DMA2D_Handle);
  HAL_DMA2D_ConfigLayer(&DMA2D_Handle, 1);

  memcpy(L8_CLUT, colors, 256 * 4);
  clut.pCLUT = (uint32_t *)L8_CLUT; //(uint32_t *)colors;
  clut.CLUTColorMode = DMA2D_CCM_ARGB8888;
  clut.Size = 0xFF;

#ifdef CORE_CM7
  SCB_CleanInvalidateDCache();
  SCB_InvalidateICache();
  //SCB_InvalidateDCache_by_Addr(clut.pCLUT, clut.Size);
#endif

  HAL_DMA2D_CLUTLoad(&DMA2D_Handle, clut, 1);
  HAL_DMA2D_PollForTransfer(&DMA2D_Handle, 100);
}

// ---------------------------------------------------------------------------
// Joystick and button pin assignments
//   A0  X-axis analog   0-1023, center ~512 (right=high, left=low)
//   A1  Y-axis analog   0-1023, center ~512 (down=high,  up=low)
//   A2  Joystick SW     active LOW (INPUT_PULLUP): pressed = LOW → KEY_RSHIFT (run)
//   A3  Fire button     active LOW (INPUT_PULLUP): pressed = LOW → KEY_FIRE
//   A4  Use button      active LOW (INPUT_PULLUP): pressed = LOW → KEY_USE
//   A5  Enter button    active LOW (INPUT_PULLUP): pressed = LOW → KEY_ENTER
//
// All buttons use digitalRead + INPUT_PULLUP so a disconnected/floating
// pin reads a stable HIGH and never triggers a spurious keypress.
//
// The joystick axes use a two-consecutive-reads debounce: a direction is
// only accepted if two successive polls agree. A floating/disconnected
// axis produces noisy inconsistent readings and stays at 0 (center).
//
// analogRead on mbed H7 is slow (~1ms each). DG_GetKey is called in a
// tight loop by Doom, so we cache all input state and only refresh it
// every MY_JOY_MS milliseconds. DG_GetKey returns cached values instantly.
// ---------------------------------------------------------------------------

#define MY_JOY_MS   50      // input poll interval in ms

// ---------------------------------------------------------------------------
// Each input generates up to two events per poll: pressed=1 and pressed=0.
// We store them in a small queue so DG_GetKey can drain them one at a time.
// This gives Doom clean press AND release events, which fixes:
//   - Fire auto-firing until empty (was missing release)
//   - Enter not working for menu exit (needed clean press+release cycle)
// ---------------------------------------------------------------------------

#define MY_EVENT_QUEUE_SIZE 16

struct MyKeyEvent { int pressed; unsigned char doomKey; };
static MyKeyEvent  myEventQueue[MY_EVENT_QUEUE_SIZE];
static int         myEventHead = 0;   // next slot to write
static int         myEventTail = 0;   // next slot to read

static void myEnqueue(int pressed, unsigned char doomKey)
{
    int next = (myEventHead + 1) % MY_EVENT_QUEUE_SIZE;
    if (next == myEventTail) return;  // queue full, drop event
    myEventQueue[myEventHead] = { pressed, doomKey };
    myEventHead = next;
}

// Previous poll state — used to detect transitions (press and release)
static int myLastX     = 0;
static int myLastY     = 0;
static int myLastRun   = 0;
static int myLastFire  = 0;
static int myLastUse   = 0;
static int myLastEnter = 0;

// Two-consecutive-reads debounce state for analog axes
static int myLastRawX  = 0;
static int myLastRawY  = 0;

static uint32_t myLastJoyMs = 0;

static void myUpdateJoystick()
{
    uint32_t now = millis();
    if (now - myLastJoyMs < MY_JOY_MS) return;
    myLastJoyMs = now;

    // --- Analog axes: two-consecutive-reads debounce ---
    int vx   = analogRead(A0);
    int vy   = analogRead(A1);
    int newX = (vx > 750) ? 1 : (vx < 250) ? -1 : 0;
    int newY = (vy > 750) ? 1 : (vy < 250) ? -1 : 0;
    int curX = (newX == myLastRawX) ? newX : 0;
    int curY = (newY == myLastRawY) ? newY : 0;
    myLastRawX = newX;
    myLastRawY = newY;

    // --- Buttons: digitalRead + INPUT_PULLUP ---
    int curRun   = (digitalRead(A2) == LOW) ? 1 : 0;
    int curFire  = (digitalRead(A3) == LOW) ? 1 : 0;
    int curUse   = (digitalRead(A4) == LOW) ? 1 : 0;
    int curEnter = (digitalRead(A5) == LOW) ? 1 : 0;

    // --- Enqueue press/release events on state transitions ---
    // X axis
    if (curX != myLastX) {
        if (myLastX != 0) myEnqueue(0, myLastX > 0 ? KEY_RIGHTARROW : KEY_LEFTARROW);
        if (curX   != 0) myEnqueue(1, curX   > 0 ? KEY_RIGHTARROW : KEY_LEFTARROW);
    }
    // Y axis
    if (curY != myLastY) {
        if (myLastY != 0) myEnqueue(0, myLastY < 0 ? KEY_UPARROW : KEY_DOWNARROW);
        if (curY   != 0) myEnqueue(1, curY   < 0 ? KEY_UPARROW : KEY_DOWNARROW);
    }
    // Buttons — press on 0→1, release on 1→0
    if (curRun   != myLastRun)   myEnqueue(curRun,   KEY_RSHIFT);
    if (curFire  != myLastFire)  myEnqueue(curFire,  KEY_FIRE);
    if (curUse   != myLastUse)   myEnqueue(curUse,   KEY_USE);
    if (curEnter != myLastEnter) myEnqueue(curEnter, KEY_ENTER);

    myLastX     = curX;
    myLastY     = curY;
    myLastRun   = curRun;
    myLastFire  = curFire;
    myLastUse   = curUse;
    myLastEnter = curEnter;
}

void DG_Init()
{
    display.begin();

    // The game renders 640x400 into a 640x480 framebuffer.
    // Clear the whole buffer to black once so unused rows don't show
    // uninitialized SDRAM garbage.
    uint32_t fb = dsi_getCurrentFrameBuffer();
    memset((void*)fb, 0, 640 * 480 * 4);

    pinMode(A2, INPUT_PULLUP);  // joystick SW  → KEY_RSHIFT (run)
    pinMode(A3, INPUT_PULLUP);  // Fire button  → KEY_FIRE
    pinMode(A4, INPUT_PULLUP);  // Use button   → KEY_USE
    pinMode(A5, INPUT_PULLUP);  // Enter button → KEY_ENTER
}

static void DMA2D_CopyBuffer(uint32_t *pSrc, uint32_t *pDst)
{
  uint32_t xPos, yPos, destination;

  /*##-1- calculate the destination transfer address  ############*/
  xPos = (display.width() - DOOMGENERIC_RESX) / 2;
  yPos = (display.height() - DOOMGENERIC_RESY) / 2;

  destination = (uint32_t)pDst; // + ((yPos * stm32_getXSize()) + xPos) * 4;

  HAL_DMA2D_PollForTransfer(&DMA2D_Handle, 100);  /* wait for the previous DMA2D transfer to ends */
  /* copy the new decoded frame to the LCD Frame buffer*/
  HAL_DMA2D_Start(&DMA2D_Handle, (uint32_t)pSrc, destination, DOOMGENERIC_RESX, DOOMGENERIC_RESY);
#if defined(CORE_CM7) && !defined(DEBUG_CM7_VIDEO) 
  HAL_DMA2D_PollForTransfer(&DMA2D_Handle, 100);  /* wait for the previous DMA2D transfer to ends */
#endif
}

int DG_GetKey(int* pressed, unsigned char* doomKey)
{
    myUpdateJoystick();  // enqueues transition events, no-op if called too soon

    // Drain one event from the queue per call.
    // Doom loops DG_GetKey until it returns 0, so the queue empties
    // cleanly each frame without stalling the game loop.
    if (myEventTail != myEventHead) {
        *pressed = myEventQueue[myEventTail].pressed;
        *doomKey = myEventQueue[myEventTail].doomKey;
        myEventTail = (myEventTail + 1) % MY_EVENT_QUEUE_SIZE;
        return 1;
    }

    return 0;
}
